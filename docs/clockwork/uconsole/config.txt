# Stop the VC firmware from probing/claiming any DSI LCD itself; the
# EwokOS fbd driver owns the whole DSI1/HVS/PV pipeline. The official
# uConsole config.txt sets this too. Note: NO vc4-fkms-v3d here — fkms
# keeps the firmware display stack alive and it would fight fbd for
# HVS/PV/PLLD.
ignore_lcd=1

# EwokOS fbd scans out of physical 0x3C100000. The VC firmware reserves
# gpu_mem MB at the top of the first 1GB (default 76MB -> carve starts at
# 0x3B400000, right on top of our framebuffer). gpu_mem=16 shrinks the
# firmware carve to 0x3F000000..0x40000000 so 0x3C100000..0x3F000000 is
# free for scan-out.
gpu_mem=16
